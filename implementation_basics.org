* 🚀 Phase 1: Database & Connection (The "DataSource")

In Spring, you'd define a ~DataSource~ bean. In Python, we create a singleton ~Engine~ using SQLAlchemy.

** File: src/db/connection.py
#+begin_src python
import os
from sqlalchemy import create_engine, text

class Database:
    """Equivalent to a Spring DataSource/Configuration class."""
    _instance = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super(Database, cls).__new__(cls)
            # DATABASE_URL is typically provided by environment variables
            url = os.getenv("DATABASE_URL", "postgresql://user:password@localhost:5432/coffeedb")
            cls._instance.engine = create_engine(url)
        return cls._instance

    def execute(self, query, params=None):
        """Wrapper for raw SQL execution, similar to JdbcTemplate.query()."""
        with self.engine.connect() as conn:
            result = conn.execute(text(query), params or {})
            conn.commit()
            return result
#+end_src

* 🗄️ Phase 2: Repository Pattern (The "@Repository")

Instead of Spring Data's magic interfaces, we'll write a explicit class using the ~Database~ helper.

** File: src/db/repository.py
#+begin_src python
from src.db.connection import Database

class CafeRepository:
    """Equivalent to a @Repository in Spring."""
    def __init__(self):
        self.db = Database()

    def get_all(self):
        """Fetches all cafes. Similar to a simple findAll()."""
        query = "SELECT id, name, location FROM cafes"
        result = self.db.execute(query)
        # Convert rows to dictionaries (like a RowMapper)
        return [dict(row._mapping) for row in result]

    def create(self, name, location):
        """Inserts a new cafe and returns its ID."""
        query = "INSERT INTO cafes (name, location) VALUES (:name, :location) RETURNING id"
        result = self.db.execute(query, {"name": name, "location": location})
        # Fetches the first row, first column
        return result.fetchone()[0]
#+end_src

* 🌐 Phase 3: API Layer (The "@RestController")

Flask routes are functional. You "autowire" the repository by instantiating it.

** File: src/app.py
#+begin_src python
import os
from dotenv import load_dotenv
from flask import Flask, jsonify, request
from src.db.repository import CafeRepository

# Load variables from .env file
load_dotenv()

app = Flask(__name__)
# Autowire the repository by instantiating it
repo = CafeRepository()

@app.route("/ping", methods=["GET"])
def ping():
    """Basic health check route."""
    return jsonify({"status": "online", "message": "Coffee Finder API is running"}), 200

@app.route("/api/cafes", methods=["GET"])
def list_cafes():
    """Returns a list of all cafes."""
    try:
        cafes = repo.get_all()
        return jsonify(cafes), 200
    except Exception as e:
        return jsonify({"error": str(e)}), 500

@app.route("/api/cafes", methods=["POST"])
def add_cafe():
    """Creates a new cafe from a JSON body."""
    data = request.json
    if not data or not data.get("name"):
        return jsonify({"error": "Name is required"}), 400
    
    try:
        cafe_id = repo.create(data['name'], data.get('location'))
        return jsonify({"id": cafe_id, "status": "created"}), 201
    except Exception as e:
        return jsonify({"error": str(e)}), 500

if __name__ == "__main__":
    # We use host='0.0.0.0' so it's accessible from outside the container later
    port = int(os.getenv("PORT", 5000))
    app.run(host="0.0.0.0", port=port, debug=True)
#+end_src

WORKDIR /app

# Install build dependencies
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    libpq-dev \
    && rm -rf /var/lib/apt/lists/*

COPY src/requirements.txt .
RUN pip install --no-cache-dir --prefix=/install -r requirements.txt

# Stage 2: Runtime
FROM python:3.12-slim

WORKDIR /app

# Install runtime dependencies for psycopg
RUN apt-get update && apt-get install -y --no-install-recommends \
    libpq5 \
    && rm -rf /var/lib/apt/lists/*

COPY --from=builder /install /usr/local
COPY . /app

# Ensure we're using the src/app.py entry point
ENV FLASK_APP=src/app.py
ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1

EXPOSE 5000

CMD ["python", "src/app.py"]
